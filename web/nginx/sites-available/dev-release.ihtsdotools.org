server {

		listen 80;
		server_name dev-release.snomedtools.org dev-release.ihtsdotools.org;
	location /api {
		proxy_set_header	Host	$host;
		proxy_set_header	X-Forwarded-For	$proxy_add_x_forwarded_for;
		proxy_pass		http://localhost:8080/api;
		proxy_redirect		http://localhost/api default;
	}

	location / {
		root /var/www/snomed-release-service-web/src/html;
		return 301 https://dev-release.ihtsdotools.org$request_uri;
	}
	
	location /cgi-bin {
		proxy_pass http://localhost:82;
	}

}
