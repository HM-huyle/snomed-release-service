server {
        listen 80;
        server_name release.snomedtools.org release.ihtsdotools.org;
	location /api {
		proxy_set_header	Host	$host;
		proxy_set_header	X-Forwarded-For	$proxy_add_x_forwarded_for;
		proxy_pass		http://localhost:8080/api;
		proxy_redirect		http://localhost/api default;
	}

        location / {
		root /var/www/snomed-release-service-web;
	}
	
	location /cgi-bin {
		proxy_pass http://localhost:82;
	}

}
server {
	listen 443;
        server_name release.ihtsdotools.org;

	ssl    on;
	ssl_certificate		/etc/ssl/certs/ihtsdotools.org.chained.crt;
	ssl_certificate_key	/etc/ssl/private/ihtsdotools.org.key;

        location /api {
                proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
                proxy_set_header Host $host;
                proxy_set_header X-Forwarded-Proto https;
                proxy_set_header X-Url-Scheme $scheme;
                proxy_pass http://localhost:8080/api;
	}

	location /axis2 {
                proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
                proxy_set_header Host $host;
                proxy_set_header X-Forwarded-Proto https;
                proxy_set_header X-Url-Scheme $scheme;
                proxy_pass http://localhost:8080/axis2;
	}

        location / {
                root /var/www/snomed-release-service-web;
        }

        location /cgi-bin {
                proxy_pass http://localhost:82;
        }

}
